1. Difference between ‘this’ and ‘let’ and when to use what, explain with examples.

The value of this is determined by how a function is called (runtime binding) where as let statement declares a block scope local variable, optionally initializing it to a value.

'this' keyword example

var test = {
prop: 42,
func: function() {
return this.prop;
},
};
console.log(test.func());
// expected output: 42

let keyword example

let x = 1;
if (x === 1) {
let x = 2;
console.log(x);
// expected output: 2
}
console.log(x);
// expected output: 1

3.Write a function to return all the keys present in an object at any level.

var object = { a: 5, b: { c: { d: 10 } } };
function iterate(obj) {
var keys = [];
for (var property in obj) {
if (obj.hasOwnProperty(property)) {
if (typeof obj[property] == "object"){
keys.push(property);
keys = keys.concat(iterate(obj[property]));
}
else {
keys.push(property)
}
}
}
return keys;
}
var result = iterate(object);
console.log(result);
Output : ["a","b","c","d"];
